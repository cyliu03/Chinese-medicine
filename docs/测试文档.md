def test_response_time():
    """测试响应时间"""
    import time
    
    predictor = TCMFormulaPredictor(
        model_path='checkpoints/best_model.pt',
        vocab_dir='data'
    )
    
    symptoms = ["恶寒", "发热", "无汗", "头痛", "身痛", "喘"]
    
    times = []
    for _ in range(100):
        start = time.time()
        predictor.predict(symptoms)
        times.append(time.time() - start)
    
    avg_time = sum(times) / len(times)
    max_time = max(times)
    min_time = min(times)
    
    print(f"平均响应时间: {avg_time*1000:.2f}ms")
    print(f"最大响应时间: {max_time*1000:.2f}ms")
    print(f"最小响应时间: {min_time*1000:.2f}ms")
    
    assert avg_time < 0.1, f"平均响应时间过长: {avg_time*1000:.2f}ms"
```

def test_concurrent_requests():
    """测试并发请求"""
    import concurrent.futures
    import requests
    
    def make_request():
        response = requests.post(
            "http://localhost:8000/predict",
            json={"symptoms": ["恶寒", "发热", "无汗"]}
        )
        return response.status_code == 200
    
    with concurrent.futures.ThreadPoolExecutor(max_workers=50) as executor:
        futures = [executor.submit(make_request) for _ in range(500)]
        results = [f.result() for f in concurrent.futures.as_completed(futures)]
    
    success_rate = sum(results) / len(results)
    print(f"成功率: {success_rate*100:.2f}%")
    
    assert success_rate > 0.99, f"成功率过低: {success_rate*100:.2f}%"
```

```python
def test_model_accuracy():
    """测试模型准确性"""
    predictor = TCMFormulaPredictor(
        model_path='checkpoints/best_model.pt',
        vocab_dir='data'
    )
    
    with open('data/test_data.json', 'r', encoding='utf-8') as f:
        test_data = json.load(f)
    
    total_tp, total_fp, total_fn = 0, 0, 0
    
    for sample in test_data:
        symptoms = sample['symptoms']
        target_herbs = set(h['name'] for h in sample['herbs'])
        
        result = predictor.predict(symptoms)
        pred_herbs = set(h['name'] for h in result['herbs'])
        
        tp = len(pred_herbs & target_herbs)
        fp = len(pred_herbs - target_herbs)
        fn = len(target_herbs - pred_herbs)
        
        total_tp += tp
        total_fp += fp
        total_fn += fn
    
    precision = total_tp / (total_tp + total_fp)
    recall = total_tp / (total_tp + total_fn)
    f1 = 2 * precision * recall / (precision + recall)
    
    print(f"Precision: {precision:.4f}")
    print(f"Recall: {recall:.4f}")
    print(f"F1 Score: {f1:.4f}")
    
    assert f1 >= 0.90, f"F1分数过低: {f1:.4f}"
```

**测试结果**：
```

Precision: 0.9703
Recall: 0.9702
F1 Score: 0.9703
```
